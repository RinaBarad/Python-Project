import matplotlib.pyplot as plt
import seaborn as sns
import pandas as pd

# Calculate percentage data for stacked bar chart
churn_counts = pd.crosstab(df["SeniorCitizen"], df["Churn"], normalize="index") * 100

# Plot
fig, ax = plt.subplots(figsize=(4,5))

# Stacked bar chart
bottom = None  # Helps in stacking
colors = ["#1f77b4", "#ff7f0e"]  # Define colors for Churn categories

for idx, churn_status in enumerate(churn_counts.columns):
    bars = ax.bar(churn_counts.index, churn_counts[churn_status], label=churn_status, 
                  bottom=bottom, color=colors[idx])
    bottom = churn_counts[churn_status] if bottom is None else bottom + churn_counts[churn_status]
    
    # Add percentage labels
    for bar in bars:
        height = bar.get_height()
        if height > 0:
            ax.text(bar.get_x() + bar.get_width()/2, bar.get_y() + height/2, 
                    f"{height:.1f}%", ha="center", va="center", fontsize=10, color="white")

# Formatting
ax.set_title("Churn by Senior Citizen")
ax.set_xlabel("Senior Citizen (0 = No, 1 = Yes)")
ax.set_ylabel("Percentage")
ax.set_ylim(0, 100)
ax.legend(title="Churn")
plt.xticks([0, 1], ["No", "Yes"])  # Replace 0 and 1 with meaningful labels

# Show plot
plt.show()

import matplotlib.pyplot as plt
import seaborn as sns

# List of categorical columns to plot
columns = [
    "PhoneService", "MultipleLines", "InternetService",
    "OnlineSecurity", "OnlineBackup", "DeviceProtection",
    "TechSupport", "StreamingTV", "StreamingMovies"
]

# Set up the figure and axes for subplots
fig, axes = plt.subplots(nrows=3, ncols=3, figsize=(15, 12))  # 3x3 grid

# Flatten the axes array for easy iteration
axes = axes.flatten()

# Loop through each column and plot
for i, col in enumerate(columns):
    sns.countplot(x=df[col], ax=axes[i], palette="viridis",hue="Churn",data=df)  # Use a nice color palette
    axes[i].set_title(f"Distribution of {col}")
    axes[i].set_xlabel("")
    axes[i].set_ylabel("Count")
    axes[i].tick_params(axis="x", rotation=45)  # Rotate x-ticks for readability

# Adjust layout
plt.tight_layout()
plt.show()

Customers who do not churn are more likely to have services like Phone Service, Internet Service (especially DSL), and Online Security activated. Higher churn rates are observed among customers who do not use or lack access to services such as Online Backup, Tech Support, and Streaming TV.
